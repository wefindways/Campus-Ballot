import { useEffect } from "react";
import $ from "jquery";
import "datatables.net-dt/js/dataTables.dataTables";
import "datatables.net-dt/css/dataTables.dataTables.css";

import PositionsRow from "./PositionsRow";

export default function PositionsTable({ positions, onEdit, onDelete }) {
  useEffect(() => {
    // Destroy old instance to avoid duplicates
    if ($.fn.DataTable.isDataTable("#positionTable")) {
      $("#positionTable").DataTable().destroy();
    }
    // Initialize new DataTable
    $("#positionTable").DataTable();

  }, [positions]); // re-run whenever rows change

  return (
    <table
      id="datatable" 
      className="table table-bordered min-w-full mt-4 border table-fixed"
    >
      <thead>
        <tr>
          <th>ID</th>
          <th>Name</th>
          <th>Email</th>
        </tr>
      </thead>
      <tbody>
        {positions.length > 0 ? (
          positions.map((pos) => (
            <PositionsRow
              key={pos.id}
              position={pos}
              onEdit={onEdit}
              onDelete={onDelete}
            />
          ))
        ) : (
          <tr>
            <td colSpan="3" className="text-center py-4 text-gray-500">
              No data found
            </td>
          </tr>
        )}
      </tbody>
    </table>
  );
}
